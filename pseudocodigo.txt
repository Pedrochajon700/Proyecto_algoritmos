    continuar <- VERDADERO

    MIENTRAS continuar = VERDADERO HACER
        IMPRIMIR "========== MENÚ PRINCIPAL =========="
        IMPRIMIR "1. Registrar nuevo curso y nota"
        IMPRIMIR "2. Mostrar todas las notas"
        IMPRIMIR "3. Calcular promedio general"
        IMPRIMIR "4. Buscar curso por nombre"
        IMPRIMIR "5. Buscar curso por código"
        IMPRIMIR "6. Ordenar cursos por promedio"
        IMPRIMIR "7. Ver historial de operaciones"
        IMPRIMIR "8. Salir"
        IMPRIMIR "===================================="
        IMPRIMIR "Ingrese una opción (1-8): "
        LEER opcion

        SI opcion = 1 ENTONCES
            <registrar_curso_y_nota>
        SINO SI opcion = 2 ENTONCES
            <mostrar_todas_las_notas>
        SINO SI opcion = 3 ENTONCES
            <calcular_promedio_general>
        SINO SI opcion = 4 ENTONCES
            <buscar_curso_por_nombre>
        SINO SI opcion = 5 ENTONCES
            <buscar_curso_por_codigo>
        SINO SI opcion = 6 ENTONCES
            <ordenar_cursos_por_promedio>
        SINO SI opcion = 7 ENTONCES
            <ver_historial_de_operaciones>
        SINO SI opcion = 8 ENTONCES
            IMPRIMIR "Saliendo del sistema..."
            continuar <- FALSO
        SINO
            IMPRIMIR "Opción no válida. Intente de nuevo."
        FIN_SI
    FIN_MIENTRAS
